<p *ngIf="user.isAdmin">Welcome, admin!</p>

<!-- @if example -->
@if (user.isAdmin) {{ '{' }}
  <p>Welcome, admin!</p>
{{ '}' }}
    


<!-- *ngIf with else -->
<ng-container *ngIf="condition; else elseBlock">
  <p>Condition is true</p>
</ng-container>
<ng-template #elseBlock>
  <p>Condition is false</p>
</ng-template>

<!-- @if with @else -->
@if (condition) {{ '{' }}
  <p>Condition is true</p>
{{ '}' }} @else {{ '{' }}
  <p>Condition is false</p>
{{ '}' }}
    


<!-- *ngFor with empty check -->
<ul>
  <li *ngFor="let item of items">{{ '{{ item }}' }}</li>
  <li *ngIf="items.length === 0">No items found</li>
</ul>

<!-- @for with @empty -->
<ul>
  @for (item of items) {{ '{' }}
    <li>{{ '{{ item }}' }}</li>
  {{ '}' }} @empty {{ '{' }}
    <li>No items found</li>
  {{ '}' }}
</ul>

<!-- *ngSwitch example -->
<div [ngSwitch]="status">
  <p *ngSwitchCase="'active'">Active</p>
  <p *ngSwitchCase="'inactive'">Inactive</p>
  <p *ngSwitchDefault>Pending</p>
</div>

<!-- @switch example -->
@switch (status) {{ '{' }}
  @case ('active') {{ '{' }} <p>Active</p> {{ '}' }}
  @case ('inactive') {{ '{' }} <p>Inactive</p> {{ '}' }}
  @default {{ '{' }} <p>Pending</p> {{ '}' }}
{{ '}' }}
