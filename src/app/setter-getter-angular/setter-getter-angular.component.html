import { NgIf } from '@angular/common';
import { Component, Input } from '@angular/core';

@Component({
  selector: 'app-selected-player',
  imports: [NgIf],
  template: `
    <div *ngIf="player">
      <h2>Tanlangan o‘yinchi: { { player.name }}</h2>
    </div>
  `
})
export class SelectedPlayerComponent {
  private _player: { name: string } | null = null;

  @Input()
  set player(value: { name: string } | null) {
    console.log('Yangi o‘yinchi:', value);
    this._player = value;
  }

  get player(): { name: string } | null {
    return this._player;
  }
}



import { Component } from '@angular/core';

@Component({
  selector: 'app-root',
  template: `
    <h1>O'yinchi tanlash</h1>
    <button (click)="selectPlayer('Messi')">Messi</button>
    <button (click)="selectPlayer('Ronaldo')">Ronaldo</button>

    <app-selected-player [player]="selectedPlayer"></app-selected-player>
  `
})
export class AppComponent {
  selectedPlayer: { name: string } | null = null;

  selectPlayer(name: string) {
    this.selectedPlayer = { name };
  }
}
